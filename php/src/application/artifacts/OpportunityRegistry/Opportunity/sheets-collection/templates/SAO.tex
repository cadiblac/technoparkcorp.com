\section*{Software Architecture Overview}

Preliminary \href{http://en.wikipedia.org/wiki/Unified_Modeling_Language}{UML} 
\href{http://en.wikipedia.org/wiki/Component_diagram}{component diagram} is
given below, which briefly outlines key software modules to be used and
developed in the project, and interconnections between them.

\newcommand{\saoComponent}[4]{
    \node [
        draw=tpcRUPBorder, 
        fill=tpcRUPBody, 
        font=\footnotesize, 
        inner sep=0mm, 
        text width=2.8cm, 
        text centered,
        minimum width=2.8cm, 
        minimum height=1.3cm, 
        anchor=center] 
        (#1) at(#2, #3) {#4};
        \node [
            draw=tpcRUPBorder, 
            below left=2mm of #1.north east, 
            anchor=north east, 
            minimum width=3mm, 
            minimum height=3mm, 
            inner sep=0mm
            ] (#1rectangle) {};
        \node [
            draw=tpcRUPBorder, 
            fill=tpcRUPBody, 
            above=0.1mm of #1rectangle.west, 
            minimum width=2.5mm, 
            minimum height=0.8mm, 
            inner sep=0mm
            ] {};
        \node [
            draw=tpcRUPBorder, 
            fill=tpcRUPBody, 
            below=0.1mm of #1rectangle.west, 
            minimum width=2.5mm, 
            minimum height=0.8mm, 
            inner sep=0mm] {};
}

\newcommand{\saoExternal}[4]{
    \node [
        draw=tpcRUPBorder, 
        fill=tpcRUPBody, 
        font=\footnotesize, 
        inner sep=0mm, 
        text width=2cm, 
        text centered,
        minimum width=2cm, 
        minimum height=1cm, 
        anchor=center] 
        (#1) at(#2, #3) {#4};
}

\newcommand{\saoBoundary}[1]{
    \node [
        draw=tpcGrey, 
        rectangle, 
        anchor=south west, 
        inner sep=0.5cm,
        fit=#1
        ] (frame) {};
    \node [
        text=tpcGrey, 
        font=\small, 
        below=0mm of frame.north east, 
        anchor=north east, 
        draw] {system.components};
}

\newcommand{\saoConnector}[3]{
    \path [
        draw=tpcBlack,
        -triangle 60,
        thick
        ] (#1) -- node [
            fill=white, 
            font={\scriptsize\setstretch{0.8}}, 
            text=tpcGrey, 
            text width=3cm, 
            inner sep=-0cm, 
            text centered
            ] {#3} (#2);
}
